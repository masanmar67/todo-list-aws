pipeline {
    agent none
    stages {
        stage('Get Code') {
            agent {label 'nodo1'}
            steps {
                // Obtener el código del repositorio en github
                echo 'Obteniendo el código de repositorio'
                git branch: 'develop', url: 'https://github.com/masanmar67/todo-list-aws.git'
                dir('config-repo') {
                    git branch: 'staging', url: 'https://github.com/masanmar67/todo-list-aws-config.git'
                }
                script {
                    sh 'cp config-repo/samconfig.toml ./'
                }
                sh 'ls'
                echo WORKSPACE
                stash includes: '**', name: 'Files'
            }
        }
        stage('Static'){
            agent {label 'nodo2'}
            steps {
                unstash 'Files'
                sh '''
                    export PATH="/home/ubuntu/.local/bin:${PATH}"
                    flake8 --format=pylint --exit-zero src > flake8.out
                '''
                recordIssues tools: [flake8(name: 'Flake8', pattern: 'flake8.out')]
            }
        }
        stage('Security'){
            agent {label 'nodo2'}
            steps {
                sh '''
                    export PATH="/home/ubuntu/.local/bin:${PATH}"
                    bandit --exit-zero -r . --format=custom -o bandit.out --msg-template "{abspath}:{line}: [{test_id}] {msg}"
                '''
                recordIssues tools: [pyLint(name: 'Bandit', pattern: 'bandit.out')]
            }
        }
        stage('Build') {
            agent {label 'nodo1'}
            steps {
                sh '''
                sam build
                '''
            }
        }
        stage('Deploy to AWS') {
            agent {label 'nodo1'}
            steps {
                sh '''
                sam deploy --config-env staging --capabilities CAPABILITY_IAM
                '''
            }
        }
        stage('Get API URL') {
            agent {label 'nodo1'}
            steps {
                script {
                    def apiUrl = sh(script: "sam list stack-outputs --stack-name todo-list-aws-staging --region us-east-1 --output json | jq -r '.[] | select(.OutputKey == \"BaseUrlApi\") | .OutputValue'", returnStdout: true).trim()
                    echo "BASE URL: ${apiUrl}"
                    writeFile file: 'base_url.txt', text: apiUrl
                    stash includes: 'base_url.txt', name: 'base_url'
                }
            }
        }
        stage('RestService'){
            agent {label 'nodo3'}
            steps {
                unstash 'Files'
                unstash 'base_url'
                script {
                    def apiUrl = readFile('base_url.txt').trim()
                    echo "API URL in RestService: ${apiUrl}"
                    withEnv(["BASE_URL=${apiUrl}"]) {
                        sh '''
                            pytest --junitxml=result-rest.xml ./test/integration
                        '''
                    }
                }
            }
        }
        stage('Promote'){
            agent {label 'nodo1'}
            steps {
                withCredentials([string(credentialsId: 'token', variable: 'GITHUB_TOKEN')]) {
                    script {
                        sh '''
                            set -e
        
                            git remote set-url origin https://${GITHUB_TOKEN}@github.com/masanmar67/todo-list-aws.git
                            git clean -f
                            git checkout master
                            git pull origin master
        
                            git merge origin/develop --no-commit --no-ff || true
        
                            # En caso de conflicto en jenkinsfile, descartamos los cambios de develop
                            if [ -f config.json ]; then
                                git checkout --ours jenkinsfile
                                git add jenkinsfile
                            fi
        
                            # Añadir el resto de archivos (sin incluir jenkinsfile)
                            git add .
        
                            # Verificamos si hay cambios para commitear
                            if ! git diff --cached --quiet; then
                                git commit -m "Merge from develop, excluding jenkinsfile"
                                git push origin master
                            else
                                echo "No changes to commit."
                            fi
                        '''
                    }
                }
            }
        }
        stage("Clean workspace1") {
            agent {label 'nodo1'}
            steps {
                script {
                    deleteDir()
                }
            }
        }
        stage("Clean workspace2") {
            agent {label 'nodo2'}
            steps {
                script {
                    deleteDir()
                }
            }
        }
        stage("Clean workspace3") {
            agent {label 'nodo3'}
            steps {
                script {
                    deleteDir()
                }
            }
        }
    }
}
